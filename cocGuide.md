# coc.nvim

curl https://bootstrap.pypa.io/get-pip.py -o get-pip.py   # 下载安装脚本

sudo python3 get-pip.py    # 运行安装脚本

pip3 install pynvim

sudo pacman -S nodejs npm

sudo npm install -g yarn

cd ~/.config/nvim/plugged/coc.nvim/

yarn install

yarn build


:checkhealth
